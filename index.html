Expandable Button Example

This is the expandable content. It will be shown when the button is clicked.

body { font-family: Arial, sans-serif; } .expand-btn { padding: 10px 20px; font-size: 16px; cursor: pointer; } .expandable-content { display: none; margin-top: 20px; padding: 10px; border: 1px solid #ccc; background-color: #f9f9f9; } document.querySelector('.expand-btn').addEventListener('click', function() { const content = document.querySelector('.expandable-content'); if (content.style.display === 'none' || content.style.display === '') { content.style.display = 'block'; } else { content.style.display = 'none'; } });
